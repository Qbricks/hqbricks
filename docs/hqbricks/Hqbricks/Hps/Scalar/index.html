and angle =
  1. | AFrac of Z.t * Z.t
    (*

    AFrac (i1, i2) represents the fraction 2Pi * i1/i2.

    *)
  2. | AFloat of Mlmpfr.mpfr_float
    (*

    Arbitrary precision float with the given value.

    *)
  3. | ArcCos of t
    (*

    Arc cosinus of the given scalar.

    *)
  4. | ArcSin of t
    (*

    Arc sinus of the given scalar.

    *)
  5. | AMulB of Hps.Hket.t * angle
    (*

    AMulB (ket, a) represents ket * a.

    *)
  6. | AMulI of Z.t * angle
    (*

    AMulI (i, a) represents i * a.

    *)
  7. | AAdd of angle * angle
    (*

    Addition.

    *)
  8. | ANeg of angle
    (*

    Negation.

    *)

Angle type represented as a fraction of 2Pi.

val fleq : ?epsilon_abs:Mlmpfr.mpfr_float -> ?epsilon_rel:Mlmpfr.mpfr_float -> t -> t -> bool option
